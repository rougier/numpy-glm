def bvec2(count = None):
    """2-components vectors of unsigned bytes (8 bits)

    Args:

        count (int):
            Number of vectors to create

    Returns:

        (np.ndarray): (count,2) shaped array with dtype np.ubyte
    """
    return ndarray.vec2(count, dtype=np.ubyte)
